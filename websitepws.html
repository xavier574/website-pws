Over Ons

Welkom bij de Bionische grijpers,Dit project is gemaakt voor mensen die een arm missen. Daarom hebben wij gekozen voor het ontwerpen en maken van een bionische hand. Met dit project laten we zien hoe technologie en creativiteit samen kunnen komen om mensen te helpen in het dagelijks leven. Dit project wordt uitgevoerd door leerlingen uit 5 havo 5 en is onderdeel van een schoolopdracht. Tijdens dit project doen wij onderzoek, ontwerpen wij oplossingen en werken wij samen aan een functioneel eindproduct. Ons doel is niet alleen om een bionische hand te maken, maar ook om andere mensen te inspireren. Wij hopen dat ons project laat zien dat je met motivatie, samenwerking en interesse in techniek je eigen projecten kunt bedenken en uitvoeren.

Adafruit_PWMServoDriver pwm = Adafruit_PWMServoDriver(0x40);

int mapFlexToServo[5] = {
  4,
  3,
  2,
  1,
  0
};

#define SERVO_OPEN  190
#define SERVO_DICHT 550

int dichtThreshold[5] = {2000, 2000, 2000, 2000, 2000};
int openThreshold[5]  = {1200, 1200, 1200, 1200, 1200};

bool invertedServo[5] = {
  false,
  false,
  true,
  true,
  true
};

bool invertedFlex[5] = {
  true,
  false,
  false,
  false,
  true
};

typedef struct {
  int flex[5];
} flexData;

flexData data;

bool fingerClosed[5] = {false, false, false, false, false};

void onReceive(const esp_now_recv_info_t *info,
               const uint8_t *incomingData,
               int len) {

  if (len != sizeof(flexData)) return;
  memcpy(&data, incomingData, sizeof(data));

  for (int s = 0; s < 5; s++) {
    int f = mapFlexToServo[s];
    int v = data.flex[f];

    if (invertedFlex[f]) {
      v = 8191 - v;
    }

    if (!fingerClosed[s] && v > dichtThreshold[f]) {
      fingerClosed[s] = true;
    }
    else if (fingerClosed[s] && v < openThreshold[f]) {
      fingerClosed[s] = false;
    }

    int pulse = fingerClosed[s] ? SERVO_DICHT : SERVO_OPEN;

    if (invertedServo[s]) {
      pulse = SERVO_OPEN + SERVO_DICHT - pulse;
    }

    pwm.setPWM(s, 0, pulse);
  }

  Serial.print("Flex: ");
  for (int i = 0; i < 5; i++) {
    Serial.print(data.flex[i]);
    Serial.print("\t");
  }
  Serial.println();
}

void setup() {
  Serial.begin(115200);

  Wire.begin(22, 21);
  pwm.begin();
  pwm.setPWMFreq(50);

  WiFi.mode(WIFI_STA);
  WiFi.disconnect();

  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init fout");
    while (true) delay(1000);
  }

  esp_now_register_recv_cb(onReceive);

  Serial.println("ESP-NOW ontvanger klaar");
}

void loop() {
}

uint8_t receiverMAC[] = {0x08, 0xD1, 0xF9, 0xD2, 0x3C, 0x54};

int flexPins[5] = {1, 2, 3, 4, 5};

typedef struct {
  uint8_t finger[5];
} flexData;

flexData data;

int dichtThreshold[5] = {600, 650, 550, 650, 600};
int openThreshold[5]  = {1200, 1250, 1150, 1300, 1200};

bool state[5] = {false, false, false, false, false};

unsigned long lastSend = 0;
const int sendInterval = 20;

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);

  esp_now_init();

  esp_now_peer_info_t peerInfo = {};
  memcpy(peerInfo.peer_addr, receiverMAC, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;
  esp_now_add_peer(&peerInfo);
}

void loop() {
  if (millis() - lastSend < sendInterval) return;
  lastSend = millis();

  for (int i = 0; i < 5; i++) {
    int val = analogRead(flexPins[i]);

    if (!state[i] && val < dichtThreshold[i]) {
      state[i] = true;
    }
    else if (state[i] && val > openThreshold[i]) {
      state[i] = false;
    }

    data.finger[i] = state[i];
  }

  esp_now_send(receiverMAC, (uint8_t*)&data, sizeof(data));
}

}

uint8_t receiverMAC[] = {0x08, 0xD1, 0xF9, 0xD2, 0x3C, 0x54};

typedef struct {
  int flex[5];
} flexData;

flexData data;
String inputString = "";
bool stringComplete = false;

void setup() {
  Serial.begin(115200);
  WiFi.mode(WIFI_STA);

  if (esp_now_init() != ESP_OK) {
    Serial.println("ESP-NOW init failed");
    while (true) delay(1000);
  }

  esp_now_peer_info_t peerInfo = {};
  memcpy(peerInfo.peer_addr, receiverMAC, 6);
  peerInfo.channel = 0;
  peerInfo.encrypt = false;

  esp_now_add_peer(&peerInfo);

  Serial.println("ESP-NOW sender ready");
}

void loop() {
  while (Serial.available()) {
    char c = (char)Serial.read();
    if (c == '\n') {
      stringComplete = true;
    } else {
      inputString += c;
    }
  }

  if (stringComplete) {
    inputString.trim();
    inputString.replace("[", "");
    inputString.replace("]", "");

    int index = 0;
    int startPos = 0;
    int commaPos = inputString.indexOf(',');

    while (commaPos >= 0 && index < 5) {
      String valStr = inputString.substring(startPos, commaPos);
      data.flex[index] = valStr.toInt();
      startPos = commaPos + 1;
      commaPos = inputString.indexOf(',', startPos);
      index++;
    }

    if (index < 5) {
      String valStr = inputString.substring(startPos);
      data.flex[index] = valStr.toInt();
    }

    esp_now_send(receiverMAC, (uint8_t*)&data, sizeof(data));

    Serial.print("Verstuurd via ESP-NOW: ");
    for (int i = 0; i < 5; i++) {
      Serial.print(data.flex[i]);
      Serial.print(" ");
    }
    Serial.println();

    inputString = "";
    stringComplete = false;
  }
}

}

    for (;;);
  }
 
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.setCursor(0, 0);
  display.println("Servo test start...");
  display.display();
  delay(1000);
}
 
void loop() {
  display.clearDisplay();
  display.setCursor(0, 0);
  display.println("Servo rechts 3s");
  display.display();
  pwm.setPWM(SERVO_CHANNEL, 0, SERVO_RIGHT);

  pwm.setPWM(SERVO_CHANNEL, 0, SERVO_STOP);
  delay(10);

}
